@Component
public class DummyAuthFilter extends OncePerRequestFilter {

    private static final String AUTH_HEADER = "X-AUTH-TOKEN";
    private static final String VALID_TOKEN = "EMPLOYEE_API_TOKEN";

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain filterChain)
            throws ServletException, IOException {

        // Allow swagger & images without auth
        String path = request.getRequestURI();
        if (path.startsWith("/swagger") || path.startsWith("/v3") || path.startsWith("/photo")) {
            filterChain.doFilter(request, response);
            return;
        }

        String token = request.getHeader(AUTH_HEADER);

        if (token == null || !VALID_TOKEN.equals(token)) {
            response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
            response.setContentType("application/json");
            response.getWriter().write("{\"message\":\"Invalid or missing token\"}");
            return;
        }

        filterChain.doFilter(request, response);
    }
}












@Configuration
public class FilterConfig {

    @Bean
    public FilterRegistrationBean<DummyAuthFilter> authFilter(DummyAuthFilter filter) {
        FilterRegistrationBean<DummyAuthFilter> registration = new FilterRegistrationBean<>();
        registration.setFilter(filter);
        registration.addUrlPatterns("/api/*"); // protect APIs
        registration.setOrder(1);
        return registration;
    }
}
